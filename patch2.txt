    <script>
        async function embedWatermark(event) {
            event.preventDefault();
            const form = event.target;
            const formData = new FormData(form);
            const submitButton = document.getElementById('embedBtn');
            const errorDiv = document.getElementById('embedError');
            const resultDiv = document.getElementById('embedResult');
            
            errorDiv.textContent = '';
            resultDiv.style.display = 'none';
            submitButton.disabled = true;
            
            try {
                const response = await fetch('/embed', {
                    method: 'POST',
                    body: formData
                });
                
                const data = await response.json();
                
                if (!response.ok) {
                    throw new Error(data.error || 'Error processing image');
                }
                
                if (data.success && data.image) {
                    document.getElementById('watermarkedImage').src = data.image;
                    document.getElementById('alphaValue').textContent = data.alpha ? 
                        `Alpha: ${data.alpha.toFixed(4)}` : '';
                    resultDiv.style.display = 'block';
                    document.getElementById('alphaInput').value = data.alpha || 0.1;
                }
            } catch (error) {
                errorDiv.textContent = error.message || 'Error processing image';
            } finally {
                submitButton.disabled = false;
            }
        }

        async function extractWatermark(event) {
            event.preventDefault();
            const form = event.target;
            const formData = new FormData(form);
            const submitButton = document.getElementById('extractBtn');
            const errorDiv = document.getElementById('extractError');
            const resultDiv = document.getElementById('extractResult');
            
            errorDiv.textContent = '';
            resultDiv.style.display = 'none';
            submitButton.disabled = true;
            
            try {
                const response = await fetch('/extract', {
                    method: 'POST',
                    body: formData
                });
                
                const data = await response.json();
                
                if (!response.ok) {
                    throw new Error(data.error || 'Error extracting watermark');
                }
                
                if (data.success && data.image) {
                    document.getElementById('extractedImage').src = data.image;
                    resultDiv.style.display = 'block';
                }
            } catch (error) {
                errorDiv.textContent = error.message || 'Error extracting watermark';
            } finally {
                submitButton.disabled = false;
            }
        }

        function downloadImage(imageId, filename) {
            const img = document.getElementById(imageId);
            if (img && img.src) {
                const link = document.createElement('a');
                link.href = img.src;
                link.download = filename;
                link.click();
            }
        }
    </script>
